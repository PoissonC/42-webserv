server {
    listen 0.0.0.0:8001;
    server_name first.com;

    location / {
        autoindex off;
        allow_methods GET;
    }

    location /files {
        autoindex on;
        root ./files;
        allow_methods GET POST DELETE;
        return 301 http://google.com;
    }

    location /cgi {
        autoindex on;
        cgi_pass ./cgi-bin;
        allow_methods GET POST DELETE;
    }

}
